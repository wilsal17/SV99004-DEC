section .data
    num1 dd 40                                 
    num2 dd 10                                  
    msg db "Resultado: "    
    leng equ $ - msg                


section .bss
    division resb 11                        


section .text
    global _start                           


_start:
    mov eax, [num1]                           
    xor edx, edx                           
    idiv dword [num2]                           
    mov [division], eax

    
    mov ecx, 10                             
    mov edi, division + 10                     
    mov byte [edi], 0                       
    .loop:
        xor edx, edx                        
        idiv ecx                            
        add dl, '0'                         
        dec edi                           
        mov [edi], dl                       
        test eax, eax                    
        jnz .loop                         


    mov eax, 4                            
    mov ebx, 1                             
    mov ecx, msg                     
    mov edx, leng                 
    int 0x80                              

    
    mov eax, 4                             
    mov ebx, 1                            
    mov ecx, edi                           
    sub edx, edi                           
    neg edx                               
    int 0x80                             

    
    mov eax, 1                             
    int 0x80
